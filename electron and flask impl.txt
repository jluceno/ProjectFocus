jfile.js
-------------------------------------
const {app, BrowserWindow} = require('electron')

function createWindow() {
    window = new BrowserWindow({width: 800, height: 600})
    window.loadFile('hfile.html')

    var python = require('child_process').spawn('python', ['./pfile.py']);
    python.stdout.on('data', function(data){
        console.log("data: ", data.toString('utf8'));
    });

    var { PythonShell } = require('python-shell');
    PythonShell.run('pfile.py', null, function (err, results) {
        if (err) throw err;
        console.log('main.py finished.');
        console.log('results: ', results);
    });

}


app.on('ready', createWindow)

app.on('window-all-closed', () => {
    if (process.platform !== 'darwin') {
        app.quit()
    }
})

INSTALL instructions for electron and flask

install flask using this command or including dependency in requirements.txt
pip install flask

install electron using
npm install --save-dev electron

point flask to the python file
set FLASK_APP=pfile.py

make sure that the python shell var works inside jfile.js
npm install --save python-shell

to run the html doc, use
npm start
